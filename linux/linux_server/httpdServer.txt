3.安装mysql5.7
3.1安装官方yum源
rpm -ivh https://dev.mysql.com/get/mysql80-community-release-el7-1.noarch.rpm #安装mysql官方yum源
yum list |grep mysql #查看mysql的yum源:  #发现只要80包，没有57包?
3.2设置默认安装mysql5.7
yum -y install yum-utils #安装yum-config-manager
vim /etc/yum/pluginconf.d/subscription-manager.conf  #yum-config-manger报错，该系统没有在授权服务器上注册.
	[main]
	enabled=0
yum-config-manager --disable mysql80-community  #关闭80包，开启57包:
yum-config-manager --enable mysql57-community  
3.3安装mysql57
#yum list |grep  mysql #全是57
yum -y install mysql-community-server mysql-community-client 
3.4启动mysql
systemctl start mysqld
3.5查看密码
grep 'temporary password' /var/log/mysqld.log 
3.6修改密码
#取消密码复杂度设置[可选]
#set global validate_password_policy=0;
#set global validate_password_mixed_case_count=0;
#set global validate_password_number_count=3;
#set global validate_password_special_char_count=0;
#set global validate_password_length=3;	
mysql -uroot -p密码 #默认密码无法登录

vim /etc/my.cnf #添加skip-grant-tables=1 
#表示使其跳过密码验证
#重启服务
mysql -uroot -p #使用任意密码登录
#set password for root@localhost=password('新密码');
update mysql.user set authentication_string=password("admin123") where user="root" and Host ="localhost";
flush privileges
vim /etc/my.cnf #删除skip-grant-tables=1 
systemctl restart mysqld #重启，使用密码进入
set password for root@localhost=password("admin123K#"); #设置性的密码
3.7设置mysql远程登录
select host,user from mysql.user; #查看user和host字段
update mysql.user set host="%" where host="localhost" and user="root";
flush privileges;

#腾讯云中，进入轻量应用服务器,在当前实例的控制台中，进入防火墙，放行3306端口。
mysql -h 1.1.1.1 -uroot -proot #远程登录 

4.编译安装python3.7
