1.安装
参考官网安装教程安装。
systemctl  start  docker  
systemctl  enable  docker  
----不常用-----
1.1 下载rpm包到服务器上
tar zxvf docker-rpm.tar.gz  #解压
cd docker-rpm  #可以自己做一个repodata[createrepo ./  #可以生成repodata]
vim /etc/yum.repos.d/docker-rpm.repo 
	[docker-rpm]
	baseurl=file:///root/docker-rpm
	enable=1
	gpgcheck=0
yum -y install docker

1.2 安装旧版本的docker
yum install epel-release -y
yum install docker-io -y  #直接安装比较慢
