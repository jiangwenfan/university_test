hive:
1.hive安装及配置:
tar zxvf apache-hive-1.2.1-bin.tar.gz -C /opt/module
mv /opt/module/apache-hive-1.2.1-bin/ hive
cd /opt/module/hive/
mv conf/hive-env.sh.template conf/hive-env.sh
vim conf/hive-env.sh
	export HADOOP_HOME=/opt/module/hadoop-2/7.2/
	export HIVE_CONF_DIR=/opt/module/hive/conf 

2.hadoop集群配置:
start-dfs.sh   #启动hdfs
start-yarn.sh  #启动yarn
#在hdfs上创建两个目录
hadoop fs -mkdir /tmp
hadoop fs -mkdir -p /user/hive/warehouse
#修改这两个目录的权限为可写
hadoop fs -chmod g+w /tmp
hadoop fs -chmod g+w /user/hive/warehouse

3.hive的基本操作
vim /opt/module/data/stu.txt
	0100	打工人
	0101	舔狗
	0110	扎心
bin/hive #启动hive
>show databases;  #查看数据库
>use default; #使用默认数据库
>show tables; #查看默认数据库中的表
>drop table student; #删除student表
>create table student(id int,name string);  #创建一张表[不推荐]
>create table student(id int,name string) row format delimited fields terminated by '\t'; #创建一张表，数据内容以tab键为分割符
>desc student; #查看表的结构
>insert into student values(1000,"舔狗不配"); #向表中插入数据[不推荐]
>load data local inpath '/opt/module/stu.txt' into table student; #向表中复制导入本地文件系中的数据
>load data inpath '/stu1.txt' into table student; #向表中移动导入hdfs中的数据。
>select * from student; #查看表中的数据
>quit; #退出hive

4.mysql安装
4.1安装准备
rpm -qa |grep mysql #查看是否已经安装了mysql
rpm -e --nodeps mysql-libs-5.1.73-7.el6.x86_64 #卸载已安装的mysql
ls /opt/module/ 
	mysql-client.rpm
	mysql-server.rpm 
	mysql-connector-java.tar.gz 

4.2 安装mysql5.7,通过添加mysql的yum源进行安装。
/*
4.2安装mysql server 
rpm -ivh mysql-server.rpm 
cat /root/.mysql_secret #查看产生的随机密码
systemctl status mysql #查看mysql已启动。enable设置为开机启动
4.3安装mysql client
rpm -ivh mysql-client.rpm 
mysql -uroot -p随机密码
>set password=password("admin123");
exit

mysql8.0跳过验证，修改密码:
use mysql;
update user set authentication_string='' where user='root'; #置空root密码
alter user 'root@localhost' identified by 'admin123'; #修改root密码
https://learnku.com/articles/38455

4.3 修改mysql密码
set password for root@localhost=password('admin123');
flush privileges;
4.4 创建hive-site.xml文件，连接mysql
	root,password 
4.5 放置mysql的java驱动到hive/lib目录下。
登录mysql查看，以及生成了数据库。
4.6重启进入hive
