mysql基本操作:
表是数据库存放数据的对象实体。
1.创建一个数据库 
create database school; #创建一个school数据库。
#删除数据库
2.创建一个表。-->增
2.1创建一个student1表.  
create table student1(
	ID char(6) not null,primary key,
	name char(8) not null,
	gender tinyint(1) not null default 1,
	出生日期 date not null,
	总学分 tinyint(1) null,
	照片 blob null,
	备注 text null
);
#创建student1表，且指定编码
create table student1(
	........
)character set=utf8;

2.2 复制student1表结构创建Student2表: 没有复制student1中的数据。只复制结构。【like】
create table student2 like student1;
2.3 复制student1的表结构和表数据创建Student3表。结构和数据全部复制。【as】
create table student3 as (select * from student1);	
2.4删除一个表
drop table if exists test1;	
//删除表test1.这个命令将表的完整性约束，索引，以及和表相关的一切权限一并删除。

3.修改表：--->改
3.1.在student1表中增加一列classNumber。
alter table student1 add column classNumber tinyint(1) null;
3.2 在student1表中删除一列classNumber。
alter table student1 drop column classNumber;
3.3 在student1表中修改classNumber列的数据类型为bigint。
alter table student1 modify classNumber bigint not null;
3.4 修改student1表名为student10表名。
rename table student1 to student10;
rename table student1 to student10,student2 to student 20;	批量修改多个表名。
alter table student1 rename to student10;
3.5 修改student1表的编码为utf8
alter table student1 convert to character set utf8;

3.6 修改school数据库编码
alter database school character set utf8;
