eip寄存器存储的是要执行的下一条指令的地址。
#mov eip,0x004183ec mov指令不能修改eip寄存器的值。
1.
jmp 0x004183ec #eip寄存器的值只能通过jmp进行修改
#jmp short ipmsg.004183ec #jmp只修改eip寄存器，无条件修改  #short表示跳转的位置离当前位置少于128字节，就用short表示。
mov eax,0x004183ec
jmp eax 

2.call 改eip的值

call 0x004183ec  ;在跳转地址这儿加上断点,f2加断点
;jmp和call都是修改了eip的值。call还修改了栈顶的值，给栈里面压入了数据，这个数据就是调用call命令的按顺序的下一条的地址值。
;观察反汇编窗口可以发现，有的指令是1个字节，有的指令是4个字节，还有的指令是5个字节。
;call指令通过计算本条指令的长度，来计算下一个顺序执行的指令的地址。
mov eax,0x004183ec
call eax

ret ;直接返回到call指令的顺序执行的下一条指令的地址。 pop eip
